#include<stdio.h>
int main()
{
    double x[12][12],sum;
    int i,j;
    int n=1;
    char c[2];
    scanf("%c",&c);
    for(i=0;i<12;i++)
    {
        for(j=0;j<12;j++)
        {
            scanf("%lf",&x[i][j]);
        }
    }
    for(i=1;i<12;i++)
    {
        for(j=0;j<n;j++)
            sum = sum + x[i][j];
            n++;
        
    }

    if(c[0]=='S') printf("%.1lf\n",sum);
    else if(c[0]=='M') printf("%.1lf\n",sum/66.0);

    return 0;
}
